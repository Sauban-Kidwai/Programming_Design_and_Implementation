    ACCESSOR: readDataFromFile
    IMPORT: filename (String)
    EXPORT: none
    Assertion: Method to read file and Parse contents to class fields.
    Algorithm:

        try:
            theFile = OPENFILE filename
            numRows = 0
            READ Line FROM theFile
            WHILENOT (theFile = EOF)
                latitude = theFile[8]
                longitude = theFile[9]
                coordSys = theFile[10]

                numRows = numRows + 1

            ENDWHILE
            CLOSEFILE

        catch:
            IOException

    END readDataFromFile
